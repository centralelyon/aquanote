157export function updateTable() {
158 let elem = $("#table_bod");
159 elem.empty();
160
161 let cycle_table = $("#cycle_table");
162 cycle_table.empty();
163 let intermed_table = $("#intermed_table");
164 intermed_table.empty();
165 let respi_table = $("#respi_table");
166 respi_table.empty();
167 let officiels_table = $("#officiels_table");
168 officiels_table.empty();
169
170 // Vérifiez que megaData[0] et megaData[0]["lignes"] existent
171 if (megaData[0] && megaData[0]["lignes"]) {
172 let ligneKey = "ligne" + (parseInt(selected_swim) + 1).toString();
173 if (!megaData[0]["lignes"][ligneKey]) {
174 console.error(`La clé ${ligneKey} n'existe pas dans megaData[0]["lignes"]`);
175 }
176 } else {
177 console.error("megaData[0] ou megaData[0]['lignes'] est undefined");
178 document.getElementById("swimmerNameForData").innerHTML = "Nageur: Inconnu";
179 }
180 let suffixe="";
181 // Boucle à travers les nageurs
182 if (curr_swims[selected_swim]) {
183 for (let j = 0; j < curr_swims[selected_swim].length; j++) {
184 let r = curr_swims[selected_swim][j];
185 if (j==curr_swims[selected_swim].length-1){
186 suffixe=getCoteActionSuffix();
187 }
188 const allowedModes = ["enter", "end", "respi", "section", "turn", "finish", "reaction", "cycle","cycle_gauche","cycle_droite","respi_droite","respi_gauche"];
189 // modes autorisés, a modifier pour ajouter des modes ou en enlever
190 if (!allowedModes.includes(r["mode"])) {
191 r["mode"] = "cycle";
192 }
193
194
195 let mess = construct_data_row(r);
196 elem.append(mess);
197
198 // Ajouter à la table appropriée en fonction du mode
199 if (r["mode"] == "cycle") {
200 r["mode"]+=suffixe;
201 cycle_table.append(mess);
202 }
203 if (r["mode"] == "section" || r["mode"] == "end" || r["mode"] == "enter") {
204 intermed_table.append(mess);
205 }
206 if (r["mode"] == "respi") {
207 r["mode"]+=suffixe;
208 respi_table.append(mess);
209 }
210 if (r["mode"] == "turn" || r["mode"] == "finish") {
211 officiels_table.append(mess);
212 }
213 }
214 } else {
215 console.error(`curr_swims[${selected_swim}] est undefined`);
216 }
217
218 // Après avoir rempli le tableau principal
219 // Ajout du listener pour chaque ligne générée dynamiquement
220 document.querySelectorAll("#table_bod tr[data-swimmer]").forEach(row => {
221 row.addEventListener("click", function() {
222 data_onclick(
223 this.getAttribute("data-swimmer"),
224 this.getAttribute("data-event")
225 );
226 });
227 });

230function getCoteActionSuffix() {
231 const selected = document.querySelector('input[name="cote_action"]:checked');
232 if (!selected) return '';
233
234 let suffix = '';
235 switch (selected.value) {
236 case '0':
237 suffix = '_gauche';
238 break;
239 case '1':
240 suffix = ''; // neutre
241 break;
242 case '2':
243 suffix = '_droite';
244 break;
245 }
246
247 // Réinitialise à neutre
248 const neutreRadio = document.querySelector('input[name="cote_action"][value="1"]');
249 if (neutreRadio) neutreRadio.checked = true;
250
251 return suffix;
252}

262export function metrics_calculation(epreuveStyle, epreuveDistance, tempo, ampli, longueur){
263 let tempoRow = (tempo[tempo.length-1]-tempo[tempo.length-2])
264 let frequenceRow = (60.0/(tempo[tempo.length-1]-tempo[tempo.length-2]))
265 let amplitudeRow = (ampli[ampli.length-1]-ampli[ampli.length-2])
266 let vitesseRow = ((ampli[ampli.length-1]-ampli[ampli.length-2])/(tempo[tempo.length-1]-tempo[tempo.length-2])).toFixed(2)
267
268 if (epreuveStyle === "freestyle" || epreuveStyle === "dos") {
269 tempoRow *= 2;
270 frequenceRow /= 2;
271 amplitudeRow *= 2;
272 } else if (epreuveStyle === "4nages") {
273 if ((epreuveDistance === "400" && [3, 4, 7, 8].includes(longueur)) || epreuveDistance === "200" && [2, 4].includes(longueur)) {
274 return {
275 tempoRow: (tempoRow * 2).toFixed(2),
276 frequenceRow: (frequenceRow / 2).toFixed(2),
277 amplitudeRow: (amplitudeRow * 2).toFixed(2),
278 vitesseRow: vitesseRow
279 };
280 }
281 }
282
283 return {
284 tempoRow: tempoRow.toFixed(2),
285 frequenceRow: frequenceRow.toFixed(2),
286 amplitudeRow: amplitudeRow.toFixed(2),
287 vitesseRow: vitesseRow
288 };
289}
